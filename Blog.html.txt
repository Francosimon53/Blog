    <script>
        const form = document.getElementById('i485-form');
        const steps = Array.from(document.querySelectorAll('.step'));
        const sections = Array.from(document.querySelectorAll('.section'));
        const nextBtn = document.getElementById('next');
        const prevBtn = document.getElementById('prev');
        const submitBtn = document.getElementById('submit');
        const summaryGrid = document.getElementById('summary-grid');
        let current = 0;

        function updateStep(direction) {
            sections[current].hidden = true;
            steps[current].classList.remove('active');
            current = Math.max(0, Math.min(sections.length - 1, current + direction));
            sections[current].hidden = false;
            steps[current].classList.add('active');
            prevBtn.disabled = current === 0;
            nextBtn.hidden = current === sections.length - 1;
            submitBtn.hidden = current !== sections.length - 1;
        }

        function syncOtherCategory() {
            const categoria = document.getElementById('categoria');
            const otro = document.getElementById('otro');
            const isOther = categoria.value === 'Otro (especificar)';
            otro.disabled = !isOther;
            if (!isOther) {
                otro.value = '';
            }
        }

        function updateSummary() {
            const entries = {
                'Nombre': form.nombre.value,
                'Nacimiento': form.nacimiento.value,
                'País': form.pais.value,
                'Número A': form.numeroA.value,
                'Categoría': form.categoria.value,
                'Otro': form.otro.value,
                'Entrada': form.entrada.value,
                'I-94': form.i94.value,
                'Dirección': form.direccion.value,
                'Teléfono': form.telefono.value,
                'Correo': form.correo.value,
                'Firma': form.firma.value,
            };

            summaryGrid.innerHTML = Object.entries(entries)
                .map(([label, value]) => `<div class="label">${label}:</div><div>${value || '-'}</div>`)
                .join('');
        }

        form.addEventListener('input', () => {
            syncOtherCategory();
            updateSummary();
        });

        form.addEventListener('submit', (event) => {
            event.preventDefault();
            if (!form.reportValidity()) return;
            updateSummary();
            alert('Resumen generado. Revisa los datos antes de presentar el formulario oficial.');
        });

        nextBtn.addEventListener('click', () => {
            if (sections[current].querySelectorAll('input, select, textarea').length && !form.reportValidity()) {
                return;
            }
            updateStep(1);
        });

        prevBtn.addEventListener('click', () => updateStep(-1));

        document.getElementById('categoria').addEventListener('change', syncOtherCategory);

        updateSummary();
    </script>
